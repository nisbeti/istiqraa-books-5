  • [+]المقدمة
    • -وجملة ما صنعته في هذا الكتاب يمكن حصره في النقاط التالية
  • [+]باب المياه
    • -١ - الماء طاهر ما لم تتغير إحدى صفاته الثلاث (لونه - طعمه - ريحه)
    • -٢ - الماء نوعان فقط: طهور ونجس
    • -٣ - لا يجوز الاغتسال للجنب في الماء الراكد غير المتحرك
    • -٤ - يجوز للرجل أن يغتسل بما تبقى من الماء الذي اغتسلت منه المرأة
    • -٥ - لعاب الكلب نجس
    • -٦ - الماء الذي ولغ فيه الكلب ولم يتغير يعد طاهرا
    • -٧ - سؤر الهرة (لعابها) طاهر
    • -٨ - تطهير النجاسة يختلف بحسب نوعها
    • -٩ - قطع جزء من البهيمة حال حياتها حرام
    • -١٠ - جسد الميتة من الحيوانات
  • [+]باب الآنية
    • -١١ - يحرم الأكل والشراب في آنية الذهب والفضة
    • -١٢ - لا يجوز استخدام آنية الذهب والفضة حتى ولو لم يكن لأجل الشرب والأكل فيها
    • -١٣ - من توضأ بماء موضوع في آنية الذهب أو الفضة مع علمه بالتحريم فهو آثم
    • -١٤ - يجوز تضبيب الإناء إذا انكسر بشيء من الفضة
    • -١٥ - جلود الميتة تطهر بالدبغ
    • -١٦ - يجوز استعمال أواني المشركين
  • [+]باب إزالة النجاسة
    • -١٧ - لا يجوز إبقاء الخمر عند المسلم حتى ولو لم يستعملها في شرب ونحوه
    • -١٨ - الخمر نجاستها معنوية وليست حسية
    • -١٩ - بناء على ما سبق في المسألة السالفة لا حرج في استعمال الخمر في غير الشرب
    • -٢٠ - لا يجوز أكل الحمر الأهلية
    • -٢١ - سؤر الحمار (لعابه) طاهر غير نجس
    • -٢٢ - كل الحيوانات التي يجوز أكلها فلعابها طاهر
    • [+]٢٣ - ما يخرج من ذكر الرجل
      • -أ- البول: وهو نجس بالإجماع
      • -ب- الودي: وهو عصارة تخرج بعد البول على شكل نقط بيضاء، وحكمه حكم البول
      • -ج- المني: وهو سائل غليظ له رائحة ويخرج دفقا عند اشتداد الشهوة. وهو طاهر ليس بنجس
      • -د- المذي: وهو ماء رقيق ليس له رائحة كرائحة المني، ويخرج بدون دفق عند فتور الشهوة بعد اشتدادها. وهو موجب لغسل الذكر والخصيتين قبل الوضوء
    • [+]٢٤ - ما يخرج من فرج المرأة
      • -أ- البول: وهو نجس كبول الرجل
      • -ب- دم الحيض: وهو دم نجس، يمنع الصلاة والصوم
      • -ج- ماء الاحتلام: وهو كالمني بالنسبة للرجل يوجب الغسل
      • -د- السوائل الخارجة من الفرج والمسماة بالرطوبة، وهذه طاهرة ليست بنجسة شأنها شأن العرق الخارج من الجسم
      • -هـ- دم الاستحاضة: وهو دم نجس يوجب الوضوء وإذا استمر فلا يمنع الصلاة ولا الصيام
    • -٢٥ - إذا وقع بول الطفل على ثوب أو سجادة
    • -٢٦ - بول الحيوان
    • -٢٧ - الدم من حيث النجاسة والطهورية
  • [+]باب الوضوء
    • -٢٨ - السواك من سنن الوضوء: سواء قبل الشروع فيه، أو عند المضمضة
    • -٢٩ - صفة الوضوء تؤخذ من حديث حمران أن عثمان ﵁ دعا بوضوء فغسل كفيه ثلاث مرات
    • -٣٠ - غسل الكفين سنة وليس بواجب إلا إذا كان قد قام من نوم ليل فإنه لا يجوز له أن يدخل يديه في الإناء إلا بعد أن يغسلهما خارجه
    • -٣١ - المضمضة والاستنشاق واجبان في الوضوء والغسل
    • -٣٢ - المرفق داخل في غسل اليد أثناء الوضوء
    • -٣٣ - مسح الرأس مرة واحدة فقط
    • -٣٤ - الأذنان تمسحان مع الرأس
    • -٣٥ - الترتيب في فروض الوضوء واجب
    • -٣٦ - الموالاة في الوضوء هي: أن لا يؤخر المتوضئ غسل عضو إلى أن يجف الذي قبله
    • -٣٧ - لا يجوز أن يزيد المسلم في غسل أعضائه في الوضوء عن ثلاث غسلات
    • -٣٨ - الاستنثار واجب في الوضوء
    • -٣٩ - المراد بإسباغ الوضوء
    • -٤٠ - المبالغة في الاستنشاق سنة لغير الصائم
    • -٤١ - تخليل الأصابع سنة
    • -٤٢ - يجب غسل ما تحت اللحية بالماء
    • -٤٣ - أقل ما ثبت عن النبي ﷺ أنه توضأ بثلثي مد
    • -٤٤ - البدء باليمين في غسل الأعضاء هو السنة
    • -٤٥ - التسمية عند الوضوء سنة وليست بواجبة
    • -٤٦ - السنة في المضمضة والاستنشاق أن يكونا من كف واحدة
    • -٤٧ - من السنة أن يقول بعد الفراغ من الوضوء " أشهد أن لا إله إلا الله وحده لا شريك له، وأشهد أن محمدا عبده ورسوله "
    • -٤٨ - الأدعية الواردة عند غسل كل عضو لا أصل لها
  • [+]باب المسح على الخفين
    • -٤٩ - الخف: اسم لنعل مصنوع من جلود
    • -٥٠ - يشترط في صحة المسح على الخفين أن يكون قد لبسهما على طهارة مائية كاملة
    • -٥١ - مدة المسح يوم بليلته للمقيم، وثلاثة أيام بلياليهن للمسافر
    • -٥٢ - لا يقال بأفضلية الغسل للقدمين مطلقا ولا بالمسح على الخفين مطلقا
    • -٥٣ - مدة المسح تبدأ من أول مسح يعقب أول حدث
    • -٥٤ - يجوز المسح على الخف المخرق من غير تحديد لحجم الخرق
    • -٥٥ - من لبس خفا ومسح عليه قبل الحدث
    • -٥٦ - طريقة المسح على الخفين
    • -٥٧ - المسح على العمامة
    • -٥٨ - يجوز للمرأة أن تمسح على خمارها إذا شق نزعه لبرد ونحوه
    • -٥٩ - المسح على جبيرة أو لفافة مشدودة على كسر أو جرح جائز
    • -٦٠ - يختلف المسح على الجبيرة عن المسح على الخف
    • -٦١ - إذا وجد جرح أو حرق في أحد أعضاء الوضوء المستحقة للغسل
    • -٦٢ - من مسح في سفر ثم رجع إلى بلده فله حكم المقيم
  • [+]باب نواقض الوضوء
    • -٦٣ - من نام نوما مستغرقا زال معه شعوره ولم يبق إدراكه معه
    • -٦٤ - دم الحيض والاستحاضة ناقضان للوضوء
    • -٦٥ - كل خارج من السبيلين من بول أو غائط أو دم أو مذي أو ودي أو مني فيعد من نواقض الوضوء
    • -٦٦ - كل خارج من غير السبيلين لا ينقض الوضوء كالقيء والصديد والقيح والدم إلا إذا كان الذي خرج من غير السبيلين بول أو غائط
    • -٦٧ - خروج الريح أو الصوت من الدبر ناقض من نواقض الوضوء
    • -٦٨ - مس المرأة بشهوة أو بدون شهوة لا ينقض الوضوء إلا إن خرج من أحدهما شيء بسبب ذلك فينتقض الوضوء حينئذ
    • -٦٩ - من شك: هل انتقض وضوؤه أم لا فهو طاهر
    • -٧٠ - مس الذكر لا ينقض الوضوء لكن الوضوء منه مستحب
    • -٧١ - مس الأم لفرج طفلها عند تنظيفه (ذكرا أو أنثى) لا ينقض الوضوء
    • -٧٢ - أكل لحم الإبل ناقض للوضوء
    • -٧٣ - قراءة القرآن للمحدث حدثا أصغر عن ظهر قلب جائزة بالإجماع
    • -٧٤ - تغسيل الميت ومس جسده لا ينقض الوضوء
  • [+]باب آداب قضاء الحاجة
    • -٧٥ - الأولى اجتناب دخول الخلاء بشيء فيه ذكر اسم الله إلا إذا خشي على تركه خارج الخلاء
    • -٧٦ - يسن لمن دخل الخلاء أن يقدم رجله اليسرى ويقول عند دخوله: " اللهم إني أعوذ بك من الخبث والخبائث "
    • -٧٧ - الاستبراء والتنظف من الخارج من السبيلين
    • -٧٨ - الأماكن التي ينهى عن قضاء الحاجة فيها
    • -٧٩ - من آداب قضاء الحاجة
    • -٨٠ - يكره استقبال القبلة واستدبارها سواء في البنيان أو الصحراء
    • -٨١ - من استجمر فلابد من ثلاث مسحات
    • -٨٢ - لا يشترط في الاستجمار أن يكون بحجر، بل كل طاهر مباح كالخشب والخرق والمناديل ونحو ذلك
    • -٨٣ - الأشياء التي ينهى عن الاستجمار
  • [+]باب الغسل
    • -٨٤ - موجبات الغسل
    • -٨٥ - إذا نزل المني بدون دفق ولذة فلا يوجب ذلك الغسل
    • -٨٦ - المرأة إذا أسقطت ما في بطنها فإن كان قد بلغ أربعة أشهر وجب تسميته وتغسيله وتكفينه والصلاة عليه
    • -٨٧ - إذا استيقظ الإنسان فوجد بللا
    • -٨٨ - صفة الاغتسال
    • -٨٩ - الغسل بالنسبة للمرأة كالرجل تماما
    • -٩٠ - إذا دخل الكافر في الإسلام وجب عليه الاغتسال
    • -٩١ - غسل يوم الجمعة سنة مؤكدة
    • -٩٢ - يسن لمن كانت عليه جنابة وأراد أن ينام ويؤخر الغسل أن يتوضأ قبل ذلك
    • -٩٣ - المرأة إذا كان عليها غسلان كغسل الجنابة والحيض ونوت رفع الاثنين أجزأها ذلك
    • -٩٤ - لا يجوز للجنب البقاء في المسجد
    • -٩٥ - أقل ما ثبت عن النبي ﷺ أنه اغتسل بثلاثة أمداد
  • [+]باب التيمم
    • -٩٦ - التيمم هو: التعبد لله تعالى بقصد الصعيد الطيب للتمسح به على صفة مخصوصة
    • -٩٧ - التيمم رافع للحدث إلى أن يوجد الماء
    • -٩٨ - صفة التيمم الصحيحة
    • -٩٩ - من كان جنبا فاستيقظ في ليلة باردة وخشي على نفسه الضرر من الماء فعليه أن يسخن الماء
    • -١٠٠ - يصار إلى التيمم بدلا عن الماء
    • -١٠١ - من كان عنده ماء لا يكفيه إلا لبعض أعضائه وجب عليه أن يستعمله إلى أن ينتهي
    • -١٠٢ - يصح التيمم بكل ما تصاعد على وجه الأرض من تراب له غبار أو ليس له غبار، أو رمل، أو حجر
    • -١٠٣ - يجوز التيمم على الجدار إن كان مبنيا من حجر أو لبن
    • -١٠٤ - من تيمم لعدم وجود الماء وشرع في الصلاة ثم حضر الماء أثناء الصلاة
    • -١٠٥ - من دخل عليه وقت الصلاة وليس عنده ماء
    • -١٠٦ - من لم يجد ماء ولا ترابا أو رملا
  • [+]باب الحيض والاستحاضة والنفاس
    • -١٠٧ - الحيض
    • -١٠٨ - ليس هناك سن محدد في الشرع لا يأتي الحيض المرأة قبله
    • -١٠٩ - الأصل أن الحامل لا تحيض كما سبق
    • -١١٠ - لا حد لأقل الحيض ولا لأكثره
    • -١١١ - لا حد أيضا لأقل فترة للطهر بين الحيضتين
    • [+]١١٢ - يحرم بسبب وجود الحيض
      • -أ- الوطء في الفرج
      • -ب - الطلاق
      • -ج- الصلاة والصوم
    • -١١٣ - الطواف بالبيت لا يجوز في حق الحائض
    • -١١٤ - يجوز للحائض أن تقرأ القرآن من غير أن تمس المصحف
    • -١١٥ - يجوز للحائض أن تمر بالمسجد، أو تدخله لأخذ حاجة
    • -١١٦ - يعرف طهر المرأة من الحيض
    • -١١٧ - الصفرة والكدرة التي تراها النساء
    • -١١٨ - يحرم إتيان الزوجة في الفرج حال حيضها
    • -١١٩ - إذا طهرت الحائض لكنها لم تغتسل
    • -١٢٠ - الاستحاضة
    • [+]١٢١ - المستحاضة
      • -أ - المعتادة
      • -ب- المميزة
      • -ج- المتحيرة
    • -١٢٣ - لا حد لأقل النفاس
    • -١٢٤ - يثبت حكم النفاس إذا نزل الدم بسبب ما وضعته المرأة
    • -١٢٥ - الدم النازل قبل الولادة بيومين أو ثلاثة
    • -١٢٦ - لو توقف الدم مع النفاس أياما أثناء الأربعين ثم عاودها مرة أخرى
